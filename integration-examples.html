import { useState, useCallback } from 'react';

// Hook personalizado para consulta CPF
function useCPFConsulta() {
    const [loading, setLoading] = useState(false);
    const [data, setData] = useState(null);
    const [error, setError] = useState(null);

    const consultarCPF = useCallback(async (cpf) => {
        setLoading(true);
        setError(null);
        setData(null);

        try {
            // Assumindo que voc√™ importou as classes
            const api = new CPFAPI();
            const resultado = await api.consultarCPF(cpf);

            if (resultado.success) {
                setData(resultado.data);
            } else {
                setError(resultado.error);
            }
        } catch (err) {
            setError(err.message);
        } finally {
            setLoading(false);
        }
    }, []);

    return { consultarCPF, loading, data, error };
}

// Componente React
